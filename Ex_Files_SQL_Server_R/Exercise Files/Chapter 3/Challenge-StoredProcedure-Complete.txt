-- Create a stored procedure to filter countries by continentCREATE Procedure CountriesByContinent	@Continent nvarchar(30)ASEXEC sp_execute_external_script@language = N'R',@script = N'OutputDataSet <- subset(InputDataSet, Continent == X, )',@input_data_1 = N'SELECT CountryName, Continent FROM Application.Countries;',@params = N' @X nvarchar(30)',@X = @ContinentWITH RESULT SETS (([Country Name] nvarchar(60), [Continent] nvarchar(30)))-- Execute the stored procedureEXECUTE CountriesByContinent @Continent = "Europe"
